1. AWK script to display the word frequency of a text
//////////////////////////////////////////////////////////////////////////////////
awk '
{
  for (i = 1; i <= NF; i++) {
    word = $i
    gsub(/[^a-zA-Z0-9]/, "", word)
    freq[word]++
  }
}
END {
  for (w in freq) {
    print w, freq[w]
  }
}' yourfile.txt
//////////////////////////////////////////////////////////////////////////////////


2. AWK script to count repetitions of a specific string and print them in reverse
//////////////////////////////////////////////////////////////////////////////////
awk -v str="yourstring" '
{
  for (i = NF; i >= 1; i--) {
    if ($i == str) {
      count++
      rev = $i " " rev
    }
  }
}
END {
  print "Count:", count
  print "Reverse:", rev
}' yourfile.txt
//////////////////////////////////////////////////////////////////////////////////

//////////////////////////////////////////////////////////////////////////////////

5. Using sed to extract full user names from /etc/passwd
//////////////////////////////////////////////////////////////////////////////////
sed -n 's/^[^:]*:[^:]*:[^:]*:[^:]*:[^:]*\([^:]*\):.*$/\1/p' /etc/passwd
